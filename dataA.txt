#! /bin/sh

mkdir assignments &> /dev/null

while IFS='' read -r line || [[ -n "$line" ]]
do
	cd ~/assignments
	git clone $line
done < test

array=$(ls  ~/assignments)
for i in $array
do	
	counterfolder=0
	countertxt=0
	counterother=0
	counter=2
	echo $i:
	array2=$(ls ~/assignments/$i)
	url="~/assignments/$i"
	for j in $array2;
	do
		if [[ "$j" == *"."* ]];then
			echo 
		else 
			more=$(ls  ~/assignments/$i/$j)
			array2="$array2 $more"
		fi
	done
	for j in $array2;
	do
		if [[ "$j" == *"."* ]];then
			if [[ "$j" == *".txt"* ]];then
				countertxt=$((countertxt+1))
			else
				counterother=$((counterother+1))
			fi
		else 
			counterfolder=$((counterfolder+1))

		fi
	done
	
	echo other $counterother
	echo txt $countertxt
	echo folder $counterfolder

done
